# Description

有两棵$n$个点的树，编号都为$1 \dots n$。编号$i$的得分为$score_i$

你现在要选定一些编号，满足

* 相应编号的点在两棵树内都构成一个连通块
* 编号的得分之和最大

$ n \le 50$

<!-- more -->

# Solution

可以感觉到这是一个最大权闭合子图的问题，但直接计算无从依据

考虑枚举一个点$x$，将两棵树看成以$x$为根的有根树，再令连通块必须过$x$。则此时选定一个点就必须选定其在两树上父亲，跑最大权闭合子图即可

时间复杂度$O(n^4)$

# Code

```c++
#include <cstdio>
#include <cstring>
const int INF=1e9;
const int N=50+10;
inline int max(int x,int y){
	return x>y?x:y;
}
inline int min(int x,int y){
	return x<y?x:y;
}
int n;
int score[N],positiveSum;
namespace Flow{/*{{{*/
	const int N=(::N);
	const int M=(::N*3);
	int h[N],tot;
	struct Edge{
		int v,f,next;
	}e[M*2];
	int flowS,flowT;
	int dis[N],cur[N];
	void Init(){
		memset(h+1,0,sizeof(int)*flowT);
		tot=1;
	}
	void SetST(int s,int t){
		flowS=s;
		flowT=t;
	}
	void AddEdge(int u,int v,int f){
		e[++tot]=(Edge){v,f,h[u]}; h[u]=tot;
		e[++tot]=(Edge){u,0,h[v]}; h[v]=tot;
	}
	void AddEdgeFromS(int u,int f){
		AddEdge(flowS,u,f);
	}
	void AddEdgeToT(int u,int f){
		AddEdge(u,flowT,f);
	}
	bool BFS(){
		for(int i=1;i<=flowT;i++){
			dis[i]=-1;
			cur[i]=h[i];
		}
		static int q[N],head,tail;
		q[head=tail=1]=flowS;
		dis[flowS]=1;
		while(head<=tail){
			int u=q[head++];
			for(int i=h[u],v;i;i=e[i].next)
				if(e[i].f&&dis[v=e[i].v]==-1){
					dis[v]=dis[u]+1;
					if(v==flowT)
						return true;
					q[++tail]=v;
				}
		}
		return dis[flowT]!=-1;
	}
	int DFS(int u,int flow){
		if(u==flowT)
			return flow;
		int res=0,x;
		for(int i=cur[u],v;i&&flow;i=e[i].next)
			if(e[i].f&&dis[v=e[i].v]==dis[u]+1){
				x=DFS(v,min(flow,e[i].f));
				e[i].f-=x;
				e[i^1].f+=x;
				if(e[i].f)
					cur[u]=i;
				flow-=x;
				res+=x;
			}
		if(!res)
			dis[u]=-1;
		return res;
	}
	int Dinic(){
		int res=0;
		while(BFS())
			res+=DFS(flowS,INF);
		return res;
	}
}/*}}}*/
namespace Tree{
	int h1[N],h2[N],tot;
	struct Edge{
		int v,next;
	}e[N*2*2];
	void AddEdge(int u,int v,int *h){
		e[++tot]=(Edge){v,h[u]}; h[u]=tot;
		e[++tot]=(Edge){u,h[v]}; h[v]=tot;
	}
	void LinkFa(int u,int fa,int *h){
		if(fa!=0)
			Flow::AddEdge(u,fa,INF);
		for(int i=h[u],v;i;i=e[i].next)
			if((v=e[i].v)!=fa)
				LinkFa(v,u,h);
	}
}
void ReadData(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d",&score[i]);
		if(score[i]>0)
			positiveSum+=score[i];
	}
	for(int i=1;i<n;i++){
		int u,v;
		scanf("%d%d",&u,&v);
		Tree::AddEdge(u+1,v+1,Tree::h1);
	}
	for(int i=1;i<n;i++){
		int u,v;
		scanf("%d%d",&u,&v);
		Tree::AddEdge(u+1,v+1,Tree::h2);
	}
}
int Calc(int u){
	Flow::Init();
	Flow::SetST(n+1,n+2);
	for(int i=1;i<=n;i++){
		if(score[i]>=0)
			Flow::AddEdgeFromS(i,score[i]);
		else
			Flow::AddEdgeToT(i,-score[i]);
	}
	Tree::LinkFa(u,0,Tree::h1);
	Tree::LinkFa(u,0,Tree::h2);
	return positiveSum-Flow::Dinic();
}
void Solve(){
	int ans=0;
	for(int i=1;i<=n;i++)
		ans=max(ans,Calc(i));
	printf("%d\n",ans);
}
int main(){
	ReadData();
	Solve();
	return 0;
}
```
