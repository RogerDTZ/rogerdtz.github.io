# Description

　　给一个无向连通图，每条边有边权

　　你需要回答一些询问：每个询问会给出两个点$a$和$b$，请求出$a$到$b$的所有路径中经过的边的最大值的最小值


<!-- more -->
# Solution

　　好像整体二分也能暴力搞过去......

　　正解直接上最小生成树，然后直接查询$a$和$b$的树上路径最大值即可

　　正在学Kruskal重构树的做法



# Summary

　　图上路径最值问题，或者和边的极值有关的问题，考虑最小生成树



# Code

```c++
#include <cstdio>
#include <algorithm>
using namespace std;
inline void swap(int &x,int &y){
	x^=y^=x^=y;
}
inline int max(int x,int y){
	return x>y?x:y;
}
const int N=15010,M=30010;
int n,m,q;
struct Edge{
	int u,v,w;
}e[M];
void readData(){
	scanf("%d%d%d",&n,&m,&q);
	for(int i=1;i<=m;i++)
		scanf("%d%d%d",&e[i].u,&e[i].v,&e[i].w);
}
namespace T{
	int h[N],tot;
	struct Edge{
		int v,w,next;
	}e[N*2];
	void addEdge(int u,int v,int w){
		e[++tot]=(Edge){v,w,h[u]}; h[u]=tot;
		e[++tot]=(Edge){u,w,h[v]}; h[v]=tot;
	}
	int dep[N],pre[N][16],maxe[N][16];
	void dfs(int u,int fa,int faEdge){
		pre[u][0]=fa;
		dep[u]=dep[fa]+1;
		maxe[u][0]=faEdge;
		for(int i=1;i<=15;i++){
			pre[u][i]=pre[pre[u][i-1]][i-1];
			maxe[u][i]=max(maxe[u][i-1],maxe[pre[u][i-1]][i-1]);
		}
		for(int i=h[u],v;i;i=e[i].next)
			if((v=e[i].v)!=fa)
				dfs(v,u,e[i].w);
	}
	inline int query(int a,int b){
		if(dep[a]<dep[b])
			swap(a,b);
		int res=0;
		for(int i=15;i>=0;i--)
			if(dep[pre[a][i]]>=dep[b]){
				res=max(res,maxe[a][i]);
				a=pre[a][i];
			}
		if(a==b)
			return res;
		for(int i=15;i>=0;i--)
			if(pre[a][i]!=pre[b][i]){
				res=max(res,max(maxe[a][i],maxe[b][i]));
				a=pre[a][i];
				b=pre[b][i];
			}
		res=max(res,max(maxe[a][0],maxe[b][0]));
		return res;
	}
}
int bel[N];
int findBel(int x){
	return (bel[x]==x)?x:(bel[x]=findBel(bel[x]));
}
bool cmpWeight(const Edge &a,const Edge &b){
	return a.w<b.w;
}
void MST(){
	sort(e+1,e+1+m,cmpWeight);
	for(int u=1;u<=n;u++) bel[u]=u;
	for(int i=1,cur=0;i<=m&&cur<n-1;i++){
		int fu=findBel(e[i].u),fv=findBel(e[i].v);
		if(fu==fv)
			continue;
		cur++;
		bel[fu]=fv;
		T::addEdge(e[i].u,e[i].v,e[i].w);
	}
}
void answerQuery(){
	int a,b;
	for(int i=1;i<=q;i++){
		scanf("%d%d",&a,&b);
		printf("%d\n",T::query(a,b));
	}
}
int main(){
	readData();
	MST();
	T::dfs(1,0,0);
	answerQuery();
	return 0;
}
```
